AJAX = Asynchronous JavaScript et XML.
AJAX est une technique pour créer des pages web rapides et dynamiques.
AJAX permet aux pages web d'être mises à jour de manière asynchrone en échangeant de petites quantités de données avec le serveur dans les coulisses. Cela signifie qu'il est possible de mettre à jour les parties d'une page web, sans recharger la page.
Les pages web classiques qui n'utilisent pas AJAX doivent recharger la page entière si le contenu doit changer.
Exemples d'applications utilisant AJAX: Google Maps, Gmail, Youtube et onglets Facebook.

POST en JS


<body>
   <div id="reponse"></div>
   <form id="qui">
      <div>
         <label for="prenom">Prénom</label>
         <input type="text" id="prenom">
      </div>
      <div>
         <label for="nom">Nom</label>
         <input type="text" id="nom">
      </div>
      <button type="submit">Envoyer</button>
   </form>
   <script>
      document.getElementById('qui').addEventListener('submit',function(e){
         //annuler la soumission du formulaire
         e.preventDefault();
         var prenom = document.getElementById('prenom').value;
         var nom = document.getElementById('nom').value;

         //instanciation de l'objet qui permet de faire la requête en js
      var xhttp = window.XMLHttpRequest ? new XMLHttpRequest() :(window.ActiveXObject ? new ActiveXObject("Microsoft.XMLHTTP") : '');

         //on définit ce qui va être fait après l'appel AJAX
         xhttp.onreadystatechange = function(){
            if(xhttp.readyState == 4 && xhttp.status ==200){
               document.getElementById('reponse').innerHTML = xhttp.responseText;
            }
         }
         xhttp.open('GET','../hello.php?prenom='+prenom+'&nom='+nom);
         xhttp.send();
      })
   </script>
</body>

         var prenom = document.getElementById('prenom').value;
         var nom = document.getElementById('nom').value;

      var xhttp = window.XMLHttpRequest ? new XMLHttpRequest() :(window.ActiveXObject ? new ActiveXObject("Microsoft.XMLHTTP") : '');

            if(xhttp.readyState == 4 && xhttp.status ==200){

            }
         }

         xhttp.open('POST','../hello.php');

GET en jquery


<body>
   <div id="reponse"></div>
   <form id="qui">
      <div>
         <label for="prenom">Prénom</label>
         <input type="text" id="prenom">
      </div>
      <div>
         <label for="nom">Nom</label>
         <input type="text" id="nom">
      </div>
      <button type="submit">Envoyer</button>
   </form>
   <script>
      $(function(){
         $('#qui').submit(function(e){
            e.preventDefault();
            $.get(
               '../hello-json.php',
               $(this).serialize(),
               function(response){
                  $('#reponse').html(response.result);
               },
               'json'//le format de réponses reçues
            );
         });
      });
   </script>
</body>
